const CACHE_NAME = 'driver-pwa-static-v1';

const STATIC_ASSETS = [
  '/',
  '/index.html',
  '/manifest.json',
];

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(STATIC_ASSETS);
    })
  );
});

self.addEventListener('fetch', (event) => {
  const { request } = event;

  // Only cache GET + same-origin static assets
  if (request.method !== 'GET' || !request.url.startsWith(self.location.origin)) {
    return;
  }

  event.respondWith(
    caches.match(request).then((cached) => {
      return cached || fetch(request);
    })
  );
});

# Contributing to Driver Navigation PWA

Thanks for contributing to the **Driver Navigation PWA (v1)** 🎉
This document explains **how to contribute safely and consistently** to the project.

> **Important:** This codebase is governed by explicit Architectural Decision Records (ADRs).
> Contributions that violate ADRs will not be accepted.

---

## 🧭 Guiding Principles

Before writing code, understand these principles:

1. **Server is the authority**

   * Client never auto-completes deliveries
   * All critical state changes require server acknowledgment

2. **State machine is non-negotiable**

   * Delivery flow must follow ADR-026 exactly
   * UI and hooks must never bypass the state machine

3. **No silent automation**

   * The PWA does not modify Vendure fulfillment state
   * Proof is evidence, not an action trigger

4. **Foreground-only behavior (v1)**

   * No background GPS
   * No background WebSocket
   * No push notifications

If a change conflicts with these principles, it **requires a new ADR**.

---

## 📚 Required Reading (Before Contributing)

Please read these documents first:

* `ADR-024-DRIVER-NAVIGATION-PWA-ARCHITECTURE.md`
* `ADR-025-DRIVER-NAVIGATION-PWA-IMPLEMENTATION.md`
* `ADR-026-DELIVERY-STATE-MACHINE.md`
* `README.md`

These are not optional.

// src/delivery/useDeliveryWorkflow.ts

import { useDispatch, useSelector } from 'react-redux';
import {
  DeliveryStateMachine,
  DeliveryState,
} from './DeliveryStateMachine';
import { deliveryActions } from '../redux/slices/delivery.slice';
import { useLocationTracking } from '../location/useLocationTracking';
import { NavigationLauncher } from './NavigationLauncher';
import { ProofUploader } from './ProofUploader';
import { locationService } from '../location/LocationService';
import { RootState } from '../redux/store';

export function useDeliveryWorkflow() {
  const dispatch = useDispatch();
  const delivery = useSelector((s: RootState) => s.delivery);
  const state: DeliveryState = delivery.state;

  // ADR-026: GPS tracking only while en route
  useLocationTracking(DeliveryStateMachine.isEnRoute(state));

  // ---------------------------------------------------------------------------
  // State transitions
  // ---------------------------------------------------------------------------

  function startDelivery() {
    if (!delivery.deliveryId) return;

    dispatch(
      deliveryActions.setState('EN_ROUTE_TO_PICKUP'),
    );
  }

  function arriveAtPickup() {
    if (!delivery.deliveryId) return;

    dispatch(
      deliveryActions.setState('AT_PICKUP'),
    );

    // Proof required at AT_PICKUP
    dispatch(deliveryActions.setAwaitingProof(true));
  }

  function arriveAtDropoff() {
    if (!delivery.deliveryId) return;

    dispatch(
      deliveryActions.setState('AT_DROPOFF'),
    );

    // Proof required at AT_DROPOFF
    dispatch(deliveryActions.setAwaitingProof(true));
  }

  // ---------------------------------------------------------------------------
  // Proof submission (REST → WebSocket ACK)
  // ---------------------------------------------------------------------------

  async function submitProof(photo: Blob): Promise<void> {
    const proofType = DeliveryStateMachine.requiresProof(state);
    if (!proofType) return;

    if (!delivery.deliveryId) {
      console.error('[Delivery] Missing deliveryId');
      return;
    }

    // Capture GPS at proof moment (ADR-026)
    const location = await locationService.getCurrentPosition();

    await ProofUploader.upload({
      deliveryId: delivery.deliveryId,
      proofType,
      photo,
      location,
      timestamp: Date.now(),
    });

    // 🚫 DO NOT update state here
    // State advances ONLY via PROOF_ACCEPTED_V1
  }

  // ---------------------------------------------------------------------------
  // Navigation helpers
  // ---------------------------------------------------------------------------

  function navigateToPickup() {
    if (!delivery.pickup) return;
    const { lat, lon, name } = delivery.pickup;
    NavigationLauncher.launch(lat, lon, name);
  }

  function navigateToDropoff() {
    if (!delivery.dropoff) return;
    const { lat, lon, name } = delivery.dropoff;
    NavigationLauncher.launch(lat, lon, name);
  }

  return {
    delivery,
    state,
    startDelivery,
    arriveAtPickup,
    arriveAtDropoff,
    navigateToPickup,
    navigateToDropoff,
    submitProof,
  };
}

export async function compressImage(
  file: Blob,
  quality = 0.8,
  maxWidth = 1280
): Promise<Blob> {
  const img = document.createElement('img');
  img.src = URL.createObjectURL(file);

  await new Promise<void>((resolve) => {
    img.onload = () => resolve();
  });

  const scale = Math.min(1, maxWidth / img.width);
  const canvas = document.createElement('canvas');

  canvas.width = img.width * scale;
  canvas.height = img.height * scale;

  const ctx = canvas.getContext('2d')!;
  ctx.drawImage(img, 0, 0, canvas.width, canvas.height);

  return new Promise((resolve) => {
    canvas.toBlob(
      (blob) => resolve(blob!),
      'image/jpeg',
      quality
    );
  });
}

export class NavigationLauncher {
  static launch(lat: number, lon: number, label?: string) {
    const encodedLabel = label
      ? encodeURIComponent(label)
      : '';

    const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent);

    const url = isIOS
      // Apple Maps (best UX on iOS)
      ? `maps://maps.apple.com/?daddr=${lat},${lon}${encodedLabel ? `&q=${encodedLabel}` : ''}`
      // Google Maps (Android + Desktop)
      : `https://www.google.com/maps/dir/?api=1&destination=${lat},${lon}&travelmode=driving`;

    window.open(url, '_blank');
  }
}

export interface ProofUploadPayload {
  deliveryId: string;
  proofType: 'PICKUP' | 'DROPOFF';
  photo: Blob;
  location: {
    lat: number;
    lon: number;
    accuracy: number;
  };
  timestamp: number;
}

// Helper: Blob → Base64 (v1 TEMPORARY)
function blobToBase64(blob: Blob): Promise<string> {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onloadend = () => resolve(reader.result as string);
    reader.onerror = reject;
    reader.readAsDataURL(blob);
  });
}

class ProofUploaderClass {
  async upload(payload: ProofUploadPayload): Promise<void> {
    // 1️⃣ Compress image
    const compressed = await compressImage(payload.photo);

    // 2️⃣ Convert to Base64 (v1 workaround)
    const base64 = await blobToBase64(compressed);

    // 3️⃣ Map PWA proof type → backend delivery status
    const status =
      payload.proofType === 'PICKUP'
        ? 'PICKED_UP'
        : 'DELIVERED';

    // 4️⃣ PATCH existing delivery status endpoint
    await api.patch(
      `/deliveries/${payload.deliveryId}/status`,
      {
        status,
        proofUrl: base64,

        // NOTE:
        // location + timestamp intentionally omitted in v1
        // Can be added later to metadata column
      },
    );

    // 🚫 DO NOT update Redux / state here
    // ✅ State advances ONLY on WebSocket PROOF_ACCEPTED_V1
  }
}

export const ProofUploader = new ProofUploaderClass();

 * NOTE (v1):
 * proofUrl currently carries Base64 data.
 * This will be replaced by a CDN-backed URL in v2.
 */

// src/delivery/DeliveryStateMachine.ts

export type DeliveryState =
  | "ASSIGNMENT_RECEIVED"
  | "EN_ROUTE_TO_PICKUP"
  | "AT_PICKUP"
  | "EN_ROUTE_TO_DROPOFF"
  | "AT_DROPOFF"
  | "DELIVERY_COMPLETED";

export type ProofType = "PICKUP" | "DROPOFF";

const TRANSITIONS: Record<DeliveryState, DeliveryState[]> = {
  ASSIGNMENT_RECEIVED: ["EN_ROUTE_TO_PICKUP"],
  EN_ROUTE_TO_PICKUP: ["AT_PICKUP"],
  AT_PICKUP: ["EN_ROUTE_TO_DROPOFF"],
  EN_ROUTE_TO_DROPOFF: ["AT_DROPOFF"],
  AT_DROPOFF: ["DELIVERY_COMPLETED"],
  DELIVERY_COMPLETED: [],
};

export class DeliveryStateMachine {
  static canTransition(from: DeliveryState, to: DeliveryState): boolean {
    return TRANSITIONS[from]?.includes(to) ?? false;
  }

  static assertTransition(from: DeliveryState, to: DeliveryState): void {
    if (!this.canTransition(from, to)) {
      throw new Error(`Invalid delivery state transition: ${from} → ${to}`);
    }
  }

  static requiresProof(state: DeliveryState): ProofType | null {
    switch (state) {
      case "AT_PICKUP":
        return "PICKUP";
      case "AT_DROPOFF":
        return "DROPOFF";
      default:
        return null;
    }
  }

  // 🔑 Explicit ACK-driven transition
  static nextAfterProofAck(state: DeliveryState): DeliveryState | null {
    switch (state) {
      case "AT_PICKUP":
        return "EN_ROUTE_TO_DROPOFF";
      case "AT_DROPOFF":
        return "DELIVERY_COMPLETED";
      default:
        return null;
    }
  }

  static isEnRoute(state: DeliveryState): boolean {
    return state === "EN_ROUTE_TO_PICKUP" || state === "EN_ROUTE_TO_DROPOFF";
  }

  static isTerminal(state: DeliveryState): boolean {
    return state === "DELIVERY_COMPLETED";
  }
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends React.Component<
  { children: React.ReactNode },
  State
> {
  state: State = { hasError: false };

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error) {
    console.error('App crash:', error);
  }

  render() {
    if (this.state.hasError) {
      return (
        <div style={{ padding: 20 }}>
          <h2>Something went wrong</h2>
          <pre>{this.state.error?.message}</pre>
          <p>Please refresh the app.</p>
        </div>
      );
    }

    return this.props.children;
  }
}

import { io, Socket } from 'socket.io-client';
import { store } from '../redux/store';
import { websocketActions } from '../redux/slices/websocket.slice';

type SocketEventHandler = (...args: any[]) => void;

class WebSocketService {
  private socket: Socket | null = null;
  private heartbeatInterval: number | null = null;
  private lastPingTime: number = 0;
  private handlers = new Map<string, Map<SocketEventHandler, SocketEventHandler>>();

  connect(token?: string) {
    // Clean up any existing socket before creating a new one
    // This handles edge cases where socket exists but is half-open
    if (this.socket) {
      this.socket.disconnect();
      this.socket = null;
    }

    // Clear handler map on new connection
    this.handlers.clear();

    const authToken = token || store.getState().auth.token;
    if (!authToken) {
      console.warn('No auth token available for WebSocket connection');
      return;
    }

    this.socket = io(import.meta.env.VITE_DRIVER_WS_URL, {
      transports: ['websocket'],
      auth: { token: authToken },
      reconnection: true,
      reconnectionAttempts: Infinity,
      reconnectionDelay: 1000,
    });

    this.setupMetrics();
  }

  private setupMetrics() {
    if (!this.socket) return;

    // Connection events
    this.socket.on('connect', () => {
      store.dispatch(websocketActions.connected());
      this.startHeartbeat();
    });

    this.socket.on('disconnect', () => {
      store.dispatch(websocketActions.disconnected());
      this.stopHeartbeat();
    });

    // Reconnection tracking
    this.socket.on('reconnect_attempt', () => {
      store.dispatch(websocketActions.reconnectionAttempted());
    });

    // Error tracking
    this.socket.on('connect_error', (error) => {
      console.warn('WebSocket connection error:', error);
      store.dispatch(websocketActions.errorOccurred());
    });

    // Custom PONG response for latency measurement
    this.socket.on('PONG_V1', () => {
      if (this.lastPingTime) {
        const latency = Date.now() - this.lastPingTime;
        store.dispatch(websocketActions.latencyMeasured(latency));
      }
    });
  }

  private startHeartbeat() {
    this.stopHeartbeat();
    // Send a custom ping every 30 seconds to ensure connection is alive
    this.heartbeatInterval = window.setInterval(() => {
      if (this.socket?.connected) {
        this.lastPingTime = Date.now();
        this.emit('PING_V1');
        store.dispatch(websocketActions.pingSent());
      }
    }, 30000);
  }

  private stopHeartbeat() {
    if (this.heartbeatInterval !== null) {
      window.clearInterval(this.heartbeatInterval);
      this.heartbeatInterval = null;
    }
  }

  disconnect() {
    this.stopHeartbeat();
    this.socket?.disconnect();
    this.socket = null;
    this.handlers.clear();
    store.dispatch(websocketActions.disconnected());
  }

  on(event: string, handler: SocketEventHandler) {
    if (!this.socket) return;

    // Check if this exact handler is already registered for this event
    const eventHandlers = this.handlers.get(event);
    if (eventHandlers?.has(handler)) {
      // Handler already registered, don't add duplicate
      return;
    }

    const wrappedHandler: SocketEventHandler = (...args: any[]) => {
      try {
        const bytes = JSON.stringify(args).length;
        store.dispatch(websocketActions.bytesReceived(bytes));
      } catch (e) {
        // Ignore serialization errors for byte counting
      }
      return handler(...args);
    };

    if (!this.handlers.has(event)) {
      this.handlers.set(event, new Map());
    }

    this.handlers.get(event)!.set(handler, wrappedHandler);
    this.socket.on(event, wrappedHandler);
  }

  off(event: string, handler?: SocketEventHandler) {
    if (!this.socket) return;

    if (!handler) {
      // Remove all handlers for this event
      this.socket.off(event);
      this.handlers.delete(event);
      return;
    }

    // Remove specific handler if we have it tracked
    const wrapped = this.handlers.get(event)?.get(handler);
    if (wrapped) {
      this.socket.off(event, wrapped);
      // Clean up our tracking
      const eventHandlers = this.handlers.get(event);
      if (eventHandlers) {
        eventHandlers.delete(handler);
        // If no handlers left for this event, remove the event entry
        if (eventHandlers.size === 0) {
          this.handlers.delete(event);
        }
      }
    } else {
      // Handler not in our map, but still try to remove from socket.io
      // This handles edge cases where a handler was added directly
      this.socket.off(event, handler);
    }
  }

  emit(event: string, payload?: any) {
    try {
      const bytes = JSON.stringify(payload ?? null).length;
      store.dispatch(websocketActions.bytesSent(bytes));
    } catch (e) {
      // Ignore serialization errors for byte counting
    }
    this.socket?.emit(event, payload);
  }

  isConnected(): boolean {
    return !!this.socket?.connected;
  }

  // Health metrics accessor
  getMetrics() {
    return store.getState().websocket.metrics;
  }
}

export const socketService = new WebSocketService();

import { useEffect, useRef } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { locationService } from './LocationService';
import { visibilityManager } from './VisibilityManager';
import { getLocationInterval } from './batteryAwareThrottle';
import { locationActions } from '../redux/slices/location.slice';
import { socketService } from '../realtime/WebSocketService';
import { RootState } from '../redux/store';
import { store } from '../redux/store';
import { api } from '../utils/api';

export function useLocationTracking(enabled: boolean) {
  const dispatch = useDispatch();
  const paused = useSelector((s: RootState) => s.location.paused);
  const timer = useRef<number | null>(null);

  async function sendLocation() {
    try {
      const pos = await locationService.getCurrentPosition();

      const payload = {
        ...pos,
        timestamp: Date.now(),
      };

      dispatch(locationActions.locationUpdated(payload));

      // ✅ FIX: Send location to drivers endpoint (not deliveries)
      const driverId = store.getState().auth.user?.id;
      if (driverId) {
        await api.post(`/drivers/${driverId}/location`, {
          lat: pos.lat,
          lon: pos.lon,
        });
      }

      socketService.emit('LOCATION_UPDATE_V1', payload);
    } catch (err) {
      // Silent failure — manual fallback handled in UI
    }
  }

  function start() {
    if (timer.current || paused) return;

    const tick = async () => {
      await sendLocation();
      timer.current = window.setTimeout(tick, getLocationInterval());
    };

    tick();
  }

  function stop() {
    if (timer.current) {
      clearTimeout(timer.current);
      timer.current = null;
    }
  }

  useEffect(() => {
    if (!enabled) {
      stop();
      return;
    }
    start();
    return stop;
  }, [enabled, paused]);

  useEffect(() => {
    const handleVisibilityChange = (visible: boolean) => {
      if (!visible) {
        dispatch(locationActions.pauseTracking());
        stop();
      } else {
        dispatch(locationActions.resumeTracking());
        start();
      }
    };

    visibilityManager.onChange(handleVisibilityChange);

    return () => {
      visibilityManager.off(handleVisibilityChange);
    };
  }, [dispatch]);
}

type VisibilityCallback = (visible: boolean) => void;

class VisibilityManager {
  private callbacks: VisibilityCallback[] = [];

  constructor() {
    document.addEventListener('visibilitychange', () => {
      const visible = !document.hidden;
      this.callbacks.forEach(cb => cb(visible));
    });
  }

  onChange(cb: VisibilityCallback) {
    this.callbacks.push(cb);
  }

  off(cb: VisibilityCallback) {
    const index = this.callbacks.indexOf(cb);
    if (index > -1) {
      this.callbacks.splice(index, 1);
    }
  }
}

// src/location/LocationService.ts

export interface Coordinates {
  lat: number;
  lon: number;
  accuracy: number;
}

class LocationService {
  async getCurrentPosition(): Promise<Coordinates> {
    if (!navigator.geolocation) {
      throw new Error('Geolocation not supported');
    }

    return new Promise((resolve, reject) => {
      navigator.geolocation.getCurrentPosition(
        (pos) => {
          resolve({
            lat: pos.coords.latitude,
            lon: pos.coords.longitude,
            accuracy: pos.coords.accuracy,
          });
        },
        (err) => reject(err),
        {
          enableHighAccuracy: true,
          timeout: 15000,
          maximumAge: 0,
        }
      );
    });
  }

  getPollingIntervalMs(): number {
    // Default: 25s
    let interval = 25_000;

    if ('getBattery' in navigator) {
      (navigator as any).getBattery().then((battery: any) => {
        if (battery.level < 0.2) {
          interval = 45_000; // Low battery mode
        }
      });
    }

    return interval;
  }
}

import { useCallback, useRef, useEffect, useMemo, useState } from 'react';

// Hook for debouncing function calls
export const useDebounce = <T extends (...args: any[]) => any>(
  callback: T,
  delay: number
): T => {
  const timeoutRef = useRef<number | null>(null);
  const callbackRef = useRef(callback);

  // Update callback ref when callback changes
  useEffect(() => {
    callbackRef.current = callback;
  }, [callback]);

  // Create debounced function
  const debouncedCallback = useCallback(
    ((...args: Parameters<T>) => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
      
      timeoutRef.current = setTimeout(() => {
        callbackRef.current(...args);
      }, delay);
    }) as T,
    [delay]
  );

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
    };
  }, []);

  return debouncedCallback;
};

// Hook for throttling function calls
export const useThrottle = <T extends (...args: any[]) => any>(
  callback: T,
  delay: number
): T => {
  const lastCallRef = useRef<number>(0);
  const timeoutRef = useRef<number | null>(null);

  const throttledCallback = useCallback(
    ((...args: Parameters<T>) => {
      const now = Date.now();
      
      if (now - lastCallRef.current >= delay) {
        lastCallRef.current = now;
        callback(...args);
      } else {
        if (timeoutRef.current) {
          clearTimeout(timeoutRef.current);
        }
        
        timeoutRef.current = setTimeout(() => {
          lastCallRef.current = Date.now();
          callback(...args);
        }, delay - (now - lastCallRef.current));
      }
    }) as T,
    [callback, delay]
  );

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
    };
  }, []);

  return throttledCallback;
};

// Hook for virtualization support
export const useVirtualization = <T>(
  items: T[],
  itemHeight: number,
  containerHeight: number,
  overscan: number = 5
) => {
  const [scrollTop, setScrollTop] = useState(0);

  const totalHeight = items.length * itemHeight;
  
  const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan);
  const endIndex = Math.min(
    items.length - 1,
    Math.ceil((scrollTop + containerHeight) / itemHeight) + overscan
  );

  const visibleItems = items.slice(startIndex, endIndex + 1);
  const visibleStartOffset = startIndex * itemHeight;

  const handleScroll = useCallback((e: React.UIEvent<HTMLDivElement>) => {
    setScrollTop(e.currentTarget.scrollTop);
  }, []);

  return {
    totalHeight,
    visibleItems,
    visibleStartOffset,
    startIndex,
    endIndex,
    handleScroll,
  };
};

// Hook for image loading optimization
export const useImageLoader = (src: string, options?: { lazy?: boolean; placeholder?: string }) => {
  const [isLoaded, setIsLoaded] = useState(false);
  const [isError, setIsError] = useState(false);
  const [imageSrc, setImageSrc] = useState(options?.placeholder || '');

  useEffect(() => {
    const img = new Image();
    
    const handleLoad = () => {
      setIsLoaded(true);
      setImageSrc(src);
    };

    const handleError = () => {
      setIsError(true);
      setImageSrc(options?.placeholder || '');
    };

    if (options?.lazy) {
      // Intersection Observer for lazy loading
      const observer = new IntersectionObserver(
        (entries) => {
          entries.forEach((entry) => {
            if (entry.isIntersecting) {
              img.src = src;
              observer.disconnect();
            }
          });
        },
        { rootMargin: '50px' }
      );

      // We need a target element to observe, but since this is a hook,
      // we'll use a different approach for lazy loading
      const timer = setTimeout(() => {
        img.src = src;
      }, 100);

      return () => {
        clearTimeout(timer);
        observer.disconnect();
      };
    } else {
      img.src = src;
    }

    img.addEventListener('load', handleLoad);
    img.addEventListener('error', handleError);

    return () => {
      img.removeEventListener('load', handleLoad);
      img.removeEventListener('error', handleError);
    };
  }, [src, options?.lazy]);

  return { isLoaded, isError, imageSrc };
};

// Hook for performance monitoring
export const usePerformanceMonitor = (name: string) => {
  const startTimeRef = useRef<number>(0);

  const start = useCallback(() => {
    startTimeRef.current = performance.now();
  }, []);

  const end = useCallback(() => {
    if (startTimeRef.current > 0) {
      const duration = performance.now() - startTimeRef.current;
      console.log(`Performance: ${name} took ${duration.toFixed(2)}ms`);
      startTimeRef.current = 0;
      return duration;
    }
    return 0;
  }, [name]);

  const measure = useCallback(async <T>(fn: () => Promise<T> | T): Promise<T> => {
    start();
    const result = await fn();
    end();
    return result;
  }, [start, end]);

  return { start, end, measure };
};

// Hook for memory management
export const useMemoryManagement = () => {
  const cleanupRef = useRef<(() => void)[]>([]);

  const addCleanup = useCallback((cleanupFn: () => void) => {
    cleanupRef.current.push(cleanupFn);
  }, []);

  const cleanup = useCallback(() => {
    cleanupRef.current.forEach(cleanupFn => cleanupFn());
    cleanupRef.current = [];
  }, []);

  useEffect(() => {
    return cleanup;
  }, [cleanup]);

  return { addCleanup, cleanup };
};

import { useEffect, useState, useCallback } from 'react';

interface AccessibilityPreferences {
  reducedMotion: boolean;
  highContrast: boolean;
  focusVisible: boolean;
  skipLinks: boolean;
}

interface UseAccessibilityReturn {
  preferences: AccessibilityPreferences;
  toggleReducedMotion: () => void;
  toggleHighContrast: () => void;
  setFocusVisible: (enabled: boolean) => void;
  isReducedMotion: boolean;
  isHighContrast: boolean;
  isFocusVisible: boolean;
}

export const useAccessibility = (): UseAccessibilityReturn => {
  const [preferences, setPreferences] = useState<AccessibilityPreferences>({
    reducedMotion: false,
    highContrast: false,
    focusVisible: true,
    skipLinks: true,
  });

  // Check system preferences
  useEffect(() => {
    // Check for reduced motion preference
    const reducedMotionQuery = window.matchMedia('(prefers-reduced-motion: reduce)');
    const handleReducedMotionChange = (e: MediaQueryListEvent) => {
      setPreferences(prev => ({
        ...prev,
        reducedMotion: e.matches
      }));
    };
    reducedMotionQuery.addEventListener('change', handleReducedMotionChange);
    
    // Check for high contrast preference
    const highContrastQuery = window.matchMedia('(prefers-contrast: high)');
    const handleHighContrastChange = (e: MediaQueryListEvent) => {
      setPreferences(prev => ({
        ...prev,
        highContrast: e.matches
      }));
    };
    highContrastQuery.addEventListener('change', handleHighContrastChange);

    // Initial values
    setPreferences(prev => ({
      ...prev,
      reducedMotion: reducedMotionQuery.matches,
      highContrast: highContrastQuery.matches
    }));

    return () => {
      reducedMotionQuery.removeEventListener('change', handleReducedMotionChange);
      highContrastQuery.removeEventListener('change', handleHighContrastChange);
    };
  }, []);

  // Apply CSS classes based on preferences
  useEffect(() => {
    const root = document.documentElement;
    
    if (preferences.reducedMotion) {
      root.classList.add('reduced-motion');
    } else {
      root.classList.remove('reduced-motion');
    }

    if (preferences.highContrast) {
      root.classList.add('high-contrast');
    } else {
      root.classList.remove('high-contrast');
    }

    if (preferences.focusVisible) {
      root.classList.add('focus-visible-enabled');
    } else {
      root.classList.remove('focus-visible-enabled');
    }
  }, [preferences]);

  const toggleReducedMotion = useCallback(() => {
    setPreferences(prev => ({
      ...prev,
      reducedMotion: !prev.reducedMotion
    }));
  }, []);

  const toggleHighContrast = useCallback(() => {
    setPreferences(prev => ({
      ...prev,
      highContrast: !prev.highContrast
    }));
  }, []);

  const setFocusVisible = useCallback((enabled: boolean) => {
    setPreferences(prev => ({
      ...prev,
      focusVisible: enabled
    }));
  }, []);

  return {
    preferences,
    toggleReducedMotion,
    toggleHighContrast,
    setFocusVisible,
    isReducedMotion: preferences.reducedMotion,
    isHighContrast: preferences.highContrast,
    isFocusVisible: preferences.focusVisible,
  };
};

export const useFocusManagement = () => {
  const focusTrap = useCallback((container: HTMLElement | null) => {
    if (!container) return;

    const focusableElements = container.querySelectorAll(
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    ) as NodeListOf<HTMLElement>;

    const firstElement = focusableElements[0];
    const lastElement = focusableElements[focusableElements.length - 1];

    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Tab') {
        if (e.shiftKey) {
          if (document.activeElement === firstElement) {
            e.preventDefault();
            lastElement.focus();
          }
        } else {
          if (document.activeElement === lastElement) {
            e.preventDefault();
            firstElement.focus();
          }
        }
      }
    };

    container.addEventListener('keydown', handleKeyDown);
    
    return () => {
      container.removeEventListener('keydown', handleKeyDown);
    };
  }, []);

  return { focusTrap };
};

interface ThemeContextType {

  toggleTheme: () => void;
  setTheme: (theme: 'light' | 'dark') => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export const ThemeProvider = ({ children }: { children: React.ReactNode }) => {
  const [theme, setThemeState] = useState<'light' | 'dark'>('light');

  const setTheme = (newTheme: 'light' | 'dark') => {
    setThemeState(newTheme);
    localStorage.setItem('theme', newTheme);
    document.documentElement.setAttribute('data-theme', newTheme);
  };

  const toggleTheme = () => {
    const newTheme = theme === 'light' ? 'dark' : 'light';
    setTheme(newTheme);
  };

  useEffect(() => {
    const savedTheme = localStorage.getItem('theme') as 'light' | 'dark';
    const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    const initialTheme = savedTheme || (systemPrefersDark ? 'dark' : 'light');
    setTheme(initialTheme);
  }, []);

  useEffect(() => {
    const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
    const handleChange = (e: MediaQueryListEvent) => {
      if (!localStorage.getItem('theme')) {
        setTheme(e.matches ? 'dark' : 'light');
      }
    };
    mediaQuery.addEventListener('change', handleChange);
    return () => mediaQuery.removeEventListener('change', handleChange);
  }, []);

  const value: ThemeContextType = {
    theme,
    toggleTheme,
    setTheme,
  };

  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  );
};

// src/auth/TokenManager.ts

const DB_NAME = 'driver-auth-db';
const STORE_NAME = 'tokens';
const KEY = 'jwt';

export interface StoredToken {
  token: string;
  expiresAt: number;
}

class TokenManagerClass {
  private dbPromise: Promise<IDBDatabase> | null = null;

  private openDb(): Promise<IDBDatabase> {
    if (this.dbPromise) return this.dbPromise;

    this.dbPromise = new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, 1);

      request.onupgradeneeded = () => {
        const db = request.result;
        if (!db.objectStoreNames.contains(STORE_NAME)) {
          db.createObjectStore(STORE_NAME);
        }
      };

      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });

    return this.dbPromise;
  }

  async save(token: StoredToken): Promise<void> {
    const db = await this.openDb();

    await new Promise<void>((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readwrite');
      const store = tx.objectStore(STORE_NAME);

      store.put(token, KEY);

      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
      tx.onabort = () => reject(tx.error);
    });
  }

  async get(): Promise<StoredToken | null> {
    const db = await this.openDb();

    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readonly');
      const store = tx.objectStore(STORE_NAME);
      const request = store.get(KEY);

      request.onsuccess = () => {
        resolve((request.result as StoredToken) ?? null);
      };

      request.onerror = () => reject(request.error);
    });
  }

  async clear(): Promise<void> {
    const db = await this.openDb();

    await new Promise<void>((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readwrite');
      const store = tx.objectStore(STORE_NAME);

      store.delete(KEY);

      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
      tx.onabort = () => reject(tx.error);
    });
  }
}

import React from 'react';

export interface ModernButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: 'primary' | 'secondary' | 'success' | 'warning' | 'danger' | 'neutral';
  size?: 'sm' | 'md' | 'lg' | 'xl';
  icon?: React.ReactNode;
  fullWidth?: boolean;
  loading?: boolean;
  children: React.ReactNode;
}

const ModernButton: React.FC<ModernButtonProps> = ({
  variant = 'primary',
  size = 'md',
  icon,
  fullWidth = false,
  loading = false,
  children,
  className = '',
  disabled,
  onClick,
  ...props
}) => {
  // Build CSS classes
  const buttonClasses = [
    'modern-button',
    `modern-button-${variant}`,
    `modern-button-${size}`,
    fullWidth ? 'modern-button-full-width' : '',
    loading ? 'modern-button-loading' : '',
    disabled ? 'modern-button-disabled' : '',
    className
  ].filter(Boolean).join(' ');

  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    if (loading || disabled) {
      e.preventDefault();
      return;
    }
    onClick?.(e);
  };

  return (
    <button
      className={buttonClasses}
      disabled={disabled || loading}
      onClick={handleClick}
      {...props}
    >
      {/* Loading Spinner */}
      {loading && (
        <div 
          className="modern-skeleton modern-spin"
          style={{
            width: size === 'sm' ? '16px' : size === 'lg' ? '20px' : '18px',
            height: size === 'sm' ? '16px' : size === 'lg' ? '20px' : '18px',
            borderRadius: '50%',
            border: '2px solid transparent',
            borderTopColor: 'currentColor',
            marginRight: icon || children ? '0.5rem' : 0
          }}
        />
      )}

      {/* Icon */}
      {icon && !loading && (
        <span style={{ marginRight: children ? '0.5rem' : 0 }}>
          {icon}
        </span>
      )}

      {/* Text */}
      {children && (
        <span>{children}</span>
      )}
    </button>
  );
};

import { IS_DEV } from './constants';

type LogLevel = 'debug' | 'info' | 'warn' | 'error';

interface LogContext {
  [key: string]: unknown;
}

class Logger {
  private format(level: LogLevel, message: string, context?: LogContext) {
    const timestamp = new Date().toISOString();
    return {
      timestamp,
      level,
      message,
      ...(context ? { context } : {}),
    };
  }

  debug(message: string, context?: LogContext) {
    if (!IS_DEV) return;
    console.debug('[DEBUG]', this.format('debug', message, context));
  }

  info(message: string, context?: LogContext) {
    console.info('[INFO]', this.format('info', message, context));
  }

  warn(message: string, context?: LogContext) {
    console.warn('[WARN]', this.format('warn', message, context));
  }

  error(message: string, context?: LogContext) {
    console.error('[ERROR]', this.format('error', message, context));
  }
}

export const logger = new Logger();

// src/offline/IndexedDb.ts

const DB_NAME = 'driver-pwa-db';
const STORE = 'app-state';

class IndexedDb {
  private dbPromise: Promise<IDBDatabase> | null = null;

  private open(): Promise<IDBDatabase> {
    if (this.dbPromise) return this.dbPromise;

    this.dbPromise = new Promise((resolve, reject) => {
      const req = indexedDB.open(DB_NAME, 1);

      req.onupgradeneeded = () => {
        const db = req.result;
        if (!db.objectStoreNames.contains(STORE)) {
          db.createObjectStore(STORE);
        }
      };

      req.onsuccess = () => resolve(req.result);
      req.onerror = () => reject(req.error);
    });

    return this.dbPromise;
  }

  async set(key: string, value: unknown): Promise<void> {
    const db = await this.open();

    await new Promise<void>((resolve, reject) => {
      const tx = db.transaction(STORE, 'readwrite');
      const store = tx.objectStore(STORE);

      store.put(value, key);

      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
      tx.onabort = () => reject(tx.error);
    });
  }

  async get<T>(key: string): Promise<T | null> {
    const db = await this.open();

    return new Promise<T | null>((resolve, reject) => {
      const tx = db.transaction(STORE, 'readonly');
      const store = tx.objectStore(STORE);
      const request = store.get(key);

      request.onsuccess = () => {
        resolve((request.result as T) ?? null);
      };

      request.onerror = () => reject(request.error);
    });
  }

  async clear(key: string): Promise<void> {
    const db = await this.open();

    await new Promise<void>((resolve, reject) => {
      const tx = db.transaction(STORE, 'readwrite');
      const store = tx.objectStore(STORE);

      store.delete(key);

      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
      tx.onabort = () => reject(tx.error);
    });
  }
}

export const indexedDb = new IndexedDb();

/* ------------------------------------------------------------------ */
/* Subcomponents                                                        */
/* ------------------------------------------------------------------ */

function Metric({
  label,
  value,
}: {
  label: string;
  value: string | number;
}) {
  return (
    <div className="flex flex-col">
      <span className="text-xs text-gray-500">{label}</span>
      <span className="font-medium text-gray-800 truncate">
        {value}
      </span>
    </div>
  );
}

function StatusPill({ connected }: { connected: boolean }) {
  return (
    <span
      className={clsx(
        'px-2 py-0.5 rounded-full text-xs font-medium',
        connected
          ? 'bg-green-100 text-green-700'
          : 'bg-red-100 text-red-700'
      )}
    >
      {connected ? 'Connected' : 'Disconnected'}
    </span>
  );
}

/* ------------------------------------------------------------------ */
/* Helpers                                                              */
/* ------------------------------------------------------------------ */

function formatBytes(bytes: number) {
  if (bytes < 1024) return `${bytes} B`;
  if (bytes < 1024 * 1024)
    return `${(bytes / 1024).toFixed(1)} KB`;
  return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
}

function formatTime(timestamp?: number) {
  if (!timestamp) return 'Never';
  const diff = Date.now() - timestamp;
  if (diff < 1000) return 'Just now';
  if (diff < 60000) return `${Math.floor(diff / 1000)}s ago`;
  return `${Math.floor(diff / 60000)}m ago`;
}

function getHealthStatus(
  websocket: any,
  metrics: WebSocketMetrics
): string {
  if (!websocket.connected) return 'Disconnected';
  if (metrics.errorCount > 10) return 'High errors';
  if (metrics.reconnectCount > 5) return 'Unstable';
  if (metrics.latency && metrics.latency > 1000)
    return 'High latency';
  return 'Healthy';
}

/* Optional hook (unchanged behavior)                                   */
/* ------------------------------------------------------------------ */

export function useWebSocketMetrics() {
  const websocket = useSelector((state: RootState) => state.websocket);

  return {
    ...websocket,
    healthStatus: getHealthStatus(
      websocket,
      websocket.metrics
    ),
  };
}

import clsx from 'clsx';
import { ReactNode } from 'react';

interface Props extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: 'primary' | 'secondary' | 'danger';
  loading?: boolean;
}

export function Button({
  variant = 'primary',
  loading = false,
  disabled,
  children,
  className = '',
  ...props
}: Props) {
  const isDisabled = disabled || loading;

  const base =
    'w-full rounded-xl px-4 py-3 text-lg font-semibold transition flex items-center justify-center gap-2';

  const variants = {
    primary: 'bg-blue-600 text-white hover:bg-blue-700',
    secondary:
      'bg-white text-gray-900 border border-gray-300 hover:bg-gray-50',
    danger: 'bg-red-600 text-white hover:bg-red-700',
  };

  return (
    <button
      {...props}
      disabled={isDisabled}
      onClick={isDisabled ? undefined : props.onClick}
      className={clsx(
        base,
        variants[variant],
        isDisabled && 'opacity-50 cursor-not-allowed',
        className
      )}
    >
      {loading && (
        <span className="h-5 w-5 animate-spin rounded-full border-2 border-white border-t-transparent" />
      )}
      {children}
    </button>
  );
}
